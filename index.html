import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score, confusion_matrix, classification_report

plt.rcParams['font.family'] = 'DejaVu Sans'
sns.set_style('whitegrid')

np.random.seed(42)

n_samples = 200
X = pd.DataFrame({
    'age': np.random.randint(50, 85, n_samples),
    'balance_score': np.random.randint(0, 100, n_samples),
    'gait_speed': np.random.uniform(0.3, 1.5, n_samples),
    'history_of_falls': np.random.randint(0, 3, n_samples)
})

y = ((X['age'] > 70) & (X['balance_score'] < 50) | (X['history_of_falls'] > 1)).astype(int)
y = y + np.random.binomial(1, 0.1, n_samples)
y = np.clip(y, 0, 1)

print('초급: 낙상 위험 예측')
print(f'샘플 수: {len(X)}')
print(f'양성(낙상위험): {y.sum()} ({y.sum()/len(y)*100:.1f}%)')
print(f'\n데이터:')
print(X.head())

n_samples = 300
X_inter = pd.DataFrame({
    'age': np.random.randint(45, 85, n_samples),
    'balance_score': np.random.randint(0, 100, n_samples),
    'gait_speed': np.random.uniform(0.2, 1.6, n_samples),
    'history_of_falls': np.random.randint(0, 4, n_samples),
    'medication_count': np.random.randint(0, 10, n_samples),
    'mobility_score': np.random.randint(0, 100, n_samples)
})

y_inter = ((X_inter['age'] > 70) & (X_inter['balance_score'] < 50) | (X_inter['history_of_falls'] > 1)).astype(int)
y_inter = y_inter + np.random.binomial(1, 0.15, n_samples)

models = {
    'LogisticRegression': LogisticRegression(random_state=42, max_iter=1000),
    'DecisionTree': DecisionTreeClassifier(random_state=42),
    'RandomForest': RandomForestClassifier(n_estimators=100, random_state=42),
    'SVM': SVC(random_state=42, probability=True)
}

results_inter = {}
for name, model in models.items():
    cv_scores = cross_val_score(model, X_train_inter, y_train_inter, cv=5)
    model.fit(X_train_inter, y_train_inter)
    y_pred_inter = model.predict(X_test_inter)
    accuracy = accuracy_score(y_test_inter, y_pred_inter)

    results_inter[name] = {
        'CV Score': cv_scores.mean(),
        'Accuracy': accuracy,
        'Model': model
    }
    print(f'{name}: CV={cv_scores.mean():.3f}, Accuracy={accuracy:.3f}')

results_df = pd.DataFrame(results_inter).T
print(f'\n{results_df}')
print('\nRandom Forest 성능이 가장 우수합니다.')
